            // تهيئة السلايدر
            function initSlider(images) {
                const slider = document.querySelector('.image-slider');
                const sliderWrapper = slider.querySelector('.slider-wrapper');
                const prevBtn = slider.querySelector('.prev-btn');
                const nextBtn = slider.querySelector('.next-btn');
                const dotsContainer = slider.querySelector('.slider-dots');
                
                // مسح المحتوى القديم
                sliderWrapper.innerHTML = '';
                dotsContainer.innerHTML = '';
                
                let currentIndex = 0;
                
                // إنشاء صور جديدة
                images.forEach((imgSrc, index) => {
                    const img = document.createElement('img');
                    img.src = imgSrc;
                    img.alt = document.getElementById('product-title').textContent;
                    if(index === 0) img.classList.add('active');
                    sliderWrapper.appendChild(img);
                });
                
                // إنشاء نقاط التوجيه الجديدة
                images.forEach((_, index) => {
                    const dot = document.createElement('span');
                    dot.classList.add('dot');
                    if(index === 0) dot.classList.add('active');
                    dot.addEventListener('click', () => {
                        goToSlide(index);
                    });
                    dotsContainer.appendChild(dot);
                });
                
                const dots = dotsContainer.querySelectorAll('.dot');
                const imagesElements = sliderWrapper.querySelectorAll('img');
                
                function updateSlider() {
                    imagesElements.forEach((img, index) => {
                        img.classList.toggle('active', index === currentIndex);
                    });
                    
                    dots.forEach((dot, index) => {
                        dot.classList.toggle('active', index === currentIndex);
                    });
                }
                
                function goToSlide(index) {
                    currentIndex = index;
                    updateSlider();
                }
                
                function nextSlide() {
                    currentIndex = (currentIndex + 1) % images.length;
                    updateSlider();
                }
                
                function prevSlide() {
                    currentIndex = (currentIndex - 1 + images.length) % images.length;
                    updateSlider();
                }
                
                // إزالة معالجات الأحداث القديمة
                prevBtn.replaceWith(prevBtn.cloneNode(true));
                nextBtn.replaceWith(nextBtn.cloneNode(true));
                
                // إضافة معالجات الأحداث الجديدة
                slider.querySelector('.prev-btn').addEventListener('click', prevSlide);
                slider.querySelector('.next-btn').addEventListener('click', nextSlide);
                
                // التمرير التلقائي
                let slideInterval = setInterval(nextSlide, 5000);
                
                // إيقاف التمرير التلقائي عند تمرير الماوس
                slider.addEventListener('mouseenter', () => {
                    clearInterval(slideInterval);
                });
                
                slider.addEventListener('mouseleave', () => {
                    slideInterval = setInterval(nextSlide, 5000);
                });
            }

                if (selectedWilaya === "الجزائر") {
                    delivery = 300;
                } else if (["وهران", "قسنطينة", "عنابة", "البليدة", "بومرداس", "تيبازة"].includes(selectedWilaya)) {
                    delivery = 500;
                } else if (selectedWilaya) {
                    delivery = 700;
                }
                
                document.getElementById('deliveryPrice').textContent = delivery + ' د.ج';
                
                // تحديث السعر الإجمالي
                updateTotalPrice();
            });

            // تحديث السعر الإجمالي
            function updateTotalPrice() {
                const currentPrice = parseInt(document.getElementById('current-price').textContent.replace(/[^0-9]/g, ''));
                const quantity = parseInt(document.getElementById('quantity').value);
                const deliveryPrice = parseInt(document.getElementById('deliveryPrice').textContent.replace(/[^0-9]/g, ''));
                
                const totalPrice = (currentPrice * quantity) + deliveryPrice;
                document.getElementById('totalPrice').textContent = totalPrice + ' د.ج';
                document.getElementById('product-price-summary').textContent = (currentPrice * quantity) + ' د.ج';
            }

            // التحكم في الكمية
            document.getElementById('increaseQuantity').addEventListener('click', function() {
                const quantityInput = document.getElementById('quantity');
                let quantity = parseInt(quantityInput.value);
                if (quantity < 10) {
                    quantityInput.value = quantity + 1;
                    updateTotalPrice();
                }
            });

            document.getElementById('decreaseQuantity').addEventListener('click', function() {
                const quantityInput = document.getElementById('quantity');
                let quantity = parseInt(quantityInput.value);
                if (quantity > 1) {
                    quantityInput.value = quantity - 1;
                    updateTotalPrice();
                }
            });

            document.addEventListener('DOMContentLoaded', async function() {
  try {
    // 1. Chargement de la configuration
    const response = await fetch('config.json');
    if (!response.ok) throw new Error('Erreur de chargement de la configuration');
    const config = await response.json();
    window.appConfig = config;

    // 2. Initialisation des variables
    let totalOrders = config.company.initialOrders;
    let totalSales = config.company.initialSales;
    let currentProduct = 'h96-max-h313';

    // 3. Fonctions utilitaires
    function formatNumber(num) {
      return num.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    }

    function updateCounters() {
      document.getElementById('totalOrders').textContent = formatNumber(totalOrders);
      document.getElementById('totalSales').textContent = formatNumber(totalSales);
    }

    function updateTotalPrice() {
      const product = config.products[currentProduct];
      const quantity = parseInt(document.getElementById('quantity').value);
      const wilaya = document.getElementById('wilaya').value;
      
      let deliveryPrice = config.delivery.default;
      if (wilaya && config.delivery[wilaya]) {
        deliveryPrice = config.delivery[wilaya];
      }

      const totalPrice = (product.currentPrice * quantity) + deliveryPrice;
      
      document.getElementById('deliveryPrice').textContent = deliveryPrice + ' د.ج';
      document.getElementById('totalPrice').textContent = totalPrice + ' د.ج';
      document.getElementById('product-price-summary').textContent = (product.currentPrice * quantity) + ' د.ج';
    }

    // 4. Initialisation du slider
    function initSlider(images) {
      const slider = document.querySelector('.image-slider');
      const wrapper = slider.querySelector('.slider-wrapper');
      const dotsContainer = slider.querySelector('.slider-dots');
      
      wrapper.innerHTML = '';
      dotsContainer.innerHTML = '';
      
      let currentIndex = 0;
      
      images.forEach((imgSrc, index) => {
        const img = document.createElement('img');
        img.src = imgSrc;
        img.alt = config.products[currentProduct].title;
        if (index === 0) img.classList.add('active');
        wrapper.appendChild(img);
        
        const dot = document.createElement('span');
        dot.classList.add('dot');
        if (index === 0) dot.classList.add('active');
        dot.addEventListener('click', () => goToSlide(index));
        dotsContainer.appendChild(dot);
      });

      const dots = dotsContainer.querySelectorAll('.dot');
      const slides = wrapper.querySelectorAll('img');

      function goToSlide(index) {
        currentIndex = index;
        updateSlider();
      }

      function updateSlider() {
        slides.forEach((slide, i) => {
          slide.classList.toggle('active', i === currentIndex);
        });
        dots.forEach((dot, i) => {
          dot.classList.toggle('active', i === currentIndex);
        });
      }

      function nextSlide() {
        currentIndex = (currentIndex + 1) % slides.length;
        updateSlider();
      }

      function prevSlide() {
        currentIndex = (currentIndex - 1 + slides.length) % slides.length;
        updateSlider();
      }

      // Boutons de navigation
      slider.querySelector('.prev-btn').addEventListener('click', prevSlide);
      slider.querySelector('.next-btn').addEventListener('click', nextSlide);

      // Auto-slide
      let slideInterval = setInterval(nextSlide, 5000);
      slider.addEventListener('mouseenter', () => clearInterval(slideInterval));
      slider.addEventListener('mouseleave', () => slideInterval = setInterval(nextSlide, 5000));
    }

    // 5. Chargement d'un produit
    function loadProduct(productId) {
      currentProduct = productId;
      const product = config.products[productId];
      
      document.getElementById('product-title').textContent = product.title;
      document.getElementById('product-description').innerHTML = product.description;
      document.getElementById('current-price').textContent = product.currentPrice + ' د.ج';
      document.getElementById('old-price').textContent = product.oldPrice + ' د.ج';
      document.querySelector('.saving').textContent = `وفر ${product.discount}`;
      
      const featuresList = document.querySelector('#features-list ul');
      featuresList.innerHTML = '';
      product.features.forEach(feature => {
        const li = document.createElement('li');
        li.textContent = feature;
        featuresList.appendChild(li);
      });

      initSlider(product.images);
      updateTotalPrice();
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    // 6. Initialisation des événements
    document.getElementById('wilaya').addEventListener('change', function() {
      const communeSelect = document.getElementById('commune');
      communeSelect.innerHTML = '<option value="">اختر بلديتك</option>';
      
      const selectedWilaya = this.value;
      if (selectedWilaya && config.wilayas[selectedWilaya]) {
        config.wilayas[selectedWilaya].forEach(commune => {
          const option = document.createElement('option');
          option.value = commune;
          option.textContent = commune;
          communeSelect.appendChild(option);
        });
      }
      updateTotalPrice();
    });

    document.getElementById('increaseQuantity').addEventListener('click', function() {
      const qty = document.getElementById('quantity');
      if (parseInt(qty.value) < 10) qty.value = parseInt(qty.value) + 1;
      updateTotalPrice();
    });

    document.getElementById('decreaseQuantity').addEventListener('click', function() {
      const qty = document.getElementById('quantity');
      if (parseInt(qty.value) > 1) qty.value = parseInt(qty.value) - 1;
      updateTotalPrice();
    });

    document.getElementById('quantity').addEventListener('change', function() {
      let qty = parseInt(this.value);
      if (isNaN(qty) || qty < 1) qty = 1;
      if (qty > 10) qty = 10;
      this.value = qty;
      updateTotalPrice();
    });

    // 7. Gestion des produits
    document.querySelectorAll('.product-card').forEach(card => {
      card.addEventListener('click', function() {
        const productId = this.getAttribute('data-product');
        loadProduct(productId);
      });
    });

    // 8. Gestion de la commande
    document.getElementById('orderForm').addEventListener('submit', async function(e) {
      e.preventDefault();
      
      const formData = {
        fullname: document.getElementById('fullname').value,
        phone: document.getElementById('phone').value,
        wilaya: document.getElementById('wilaya').value,
        commune: document.getElementById('commune').value,
        address: document.getElementById('address').value,
        quantity: document.getElementById('quantity').value,
        product: currentProduct,
        total: document.getElementById('totalPrice').textContent
      };

      // Validation
      if (!formData.fullname || !formData.phone || !formData.wilaya || !formData.commune || !formData.address) {
        alert('الرجاء ملء جميع الحقول المطلوبة');
        return;
      }

      // Mise à jour des compteurs
      totalOrders += parseInt(formData.quantity);
      totalSales += config.products[currentProduct].currentPrice * parseInt(formData.quantity);
      updateCounters();

      // Affichage du chargement
      document.getElementById('loading').style.display = 'block';
      document.getElementById('orderForm').style.display = 'none';

      // Génération du numéro de commande
      const orderNumber = 'SE-' + Math.floor(1000 + Math.random() * 9000);
      document.getElementById('orderNumber').textContent = orderNumber;

      // Construction du message Telegram
      const product = config.products[currentProduct];
      const message = `🛒 *طلب جديد - ${product.title}* 🛒
      
👤 *الاسم الكامل*: ${formData.fullname}
📞 *رقم الهاتف*: ${formData.phone}
📍 *العنوان*: ${formData.wilaya}, ${formData.commune}, ${formData.address}
💰 *المجموع*: ${formData.total}
🔄 *الكمية*: ${formData.quantity}
📦 *المنتج*: ${product.title} (${product.currentPrice} د.ج)`;

      // Affichage du message de succès
      document.getElementById('successMessage').style.display = 'block';

      // Ouverture de Telegram
      setTimeout(() => {
        const telegramUrl = `https://t.me/${config.company.telegram}?text=${encodeURIComponent(message)}`;
        window.open(telegramUrl, '_blank');
      }, 500);
    });

    // 9. Initialisation de la page
    updateCounters();
    loadProduct(currentProduct);

    // 10. Compteur automatique
    setInterval(() => {
      const randomOrders = Math.floor(Math.random() * 3) + 1;
      totalOrders += randomOrders;
      totalSales += randomOrders * config.products[currentProduct].currentPrice;
      updateCounters();
    }, 30000);

  } catch (error) {
    console.error('Erreur:', error);
    alert('Une erreur est survenue lors du chargement de la page');
  }
});
</script>
